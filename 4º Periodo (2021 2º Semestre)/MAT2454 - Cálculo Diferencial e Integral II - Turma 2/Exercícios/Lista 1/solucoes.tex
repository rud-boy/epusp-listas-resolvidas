  &\leq 10^{-5}
  && \text{O maior erro possível ocorre quando $\bar{x} = 1$} \notag \\
  %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  3 \cdot 10^5
  &\leq (n + 1)!
  && \text{Procuramos o menor $n$ possível que satisfaça essa equação} \notag \\
  %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  \notag \\
  &
  \begin{tabular}{| c | c | r |}
    \hline

            1  &         2    &            2     \\ \hline

  \end{tabular} \notag \\
  \notag \\
  %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  P_8(1)
  &= \sum_{i = 0}^8 \frac{1}{i!}
  = 2,71827877 \dots
